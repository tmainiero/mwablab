\section{Categories}
\label{sec:category}
\concept{category}{Category}
\implements{haskell}{Cat.Category}
\implements{agda}{Cat.Category}
\implements{lisp}{category}

A \newterm{category} $\newmath{\Category{C}}$ consists of the following data
(\stacksref{0013}; \nlabref{category}):
\begin{enumerate}
  \item A collection $\Ob(\Category{C})$ of \newterm{objects}.
  \item For each pair of objects $X, Y \in \Ob(\Category{C})$,
    a set $\newmath{\Hom_{\Category{C}}(X, Y)}$ of \newterm{morphisms}
    from $X$ to $Y$.
  \item For each object $X$,
    an \newterm{identity morphism}
    $\newmath{\id_X} \in \Hom_{\Category{C}}(X, X)$.
  \item For each triple of objects $X, Y, Z$,
    a \newterm{composition map}
    \begin{equation}
      \begin{aligned}
        \circ \colon \Hom_{\Category{C}}(Y, Z) \times \Hom_{\Category{C}}(X, Y)
          &\lto \Hom_{\Category{C}}(X, Z) \\
        (g, f) &\lmapsto g \circ f
      \end{aligned}
      \label{eq:cat-comp}
    \end{equation}
\end{enumerate}

subject to the following axioms:
\begin{itemize}
  \item \textbf{Associativity.}
    \axiom{category}{associativity}
    For all morphisms $f \colon X \to Y$, $g \colon Y \to Z$, $h \colon Z \to W$,
    \begin{equation}
      \label{eq:cat-assoc}
      h \circ (g \circ f) = (h \circ g) \circ f.
    \end{equation}

  \item \textbf{Identity.}
    \axiom{category}{identity}
    For all morphisms $f \colon X \to Y$,
    \begin{equation}
      \label{eq:cat-id}
      f \circ \id_X = f = \id_Y \circ f.
    \end{equation}
\end{itemize}

\begin{remark}
The Hom functor is the assignment
\begin{equation}
  \begin{aligned}
    \Hom_{\Category{C}}(-,-) \colon
      \Category{C}^{\mathrm{op}} \times \Category{C}
      &\lto \Set \\
    (X, Y) &\lmapsto \Hom_{\Category{C}}(X, Y).
  \end{aligned}
  \label{eq:hom-functor}
\end{equation}
This will be made precise once we define \emph{functors} (\S\ref{sec:functor}).
\end{remark}
