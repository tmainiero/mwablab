\section{Natural Transformations}
\label{sec:natural-transformation}
\concept{natural-transformation}{Natural Transformation}
\depends{functor}
\implements{haskell}{Cat.NaturalTransformation}
\implements{agda}{Cat.NaturalTransformation}
\implements{lisp}{natural-transformation}

Let $\Category{C}$ and $\Category{D}$ be categories,
and let $\Functor{F}, \Functor{G} \colon \Category{C} \lto \Category{D}$ be functors.
A \newterm{natural transformation}
$\newmath{\NatTrans{\alpha} \colon \Functor{F} \Rightarrow \Functor{G}}$
consists of the following data
(\stacksref{0015}; \nlabref{natural+transformation}):
\begin{enumerate}
  \item For each object $X \in \Ob(\Category{C})$,
    a morphism $\newmath{\NatTrans{\alpha}_X} \colon \Functor{F}(X) \lto \Functor{G}(X)$
    in $\Category{D}$,
    called the \newterm{component} of $\NatTrans{\alpha}$ at $X$.
\end{enumerate}

subject to the \newterm{naturality condition}:
\axiom{natural-transformation}{naturality}
for every morphism $f \colon X \lto Y$ in $\Category{C}$,
\begin{equation}
  \label{eq:naturality}
  \Functor{G}(f) \circ \NatTrans{\alpha}_X = \NatTrans{\alpha}_Y \circ \Functor{F}(f).
\end{equation}

This is expressed by the commutativity of the \newterm{naturality square}:
\begin{equation}
  \label{eq:naturality-square}
  \begin{tikzcd}
    \Functor{F}(X) \arrow[r, "\NatTrans{\alpha}_X"] \arrow[d, "\Functor{F}(f)"']
      & \Functor{G}(X) \arrow[d, "\Functor{G}(f)"] \\
    \Functor{F}(Y) \arrow[r, "\NatTrans{\alpha}_Y"']
      & \Functor{G}(Y)
  \end{tikzcd}
\end{equation}

\begin{definition}[Identity natural transformation]
  \label{def:id-nat-trans}
  The \newterm{identity natural transformation}
  $\newmath{\id_{\Functor{F}}} \colon \Functor{F} \Rightarrow \Functor{F}$
  has components $(\id_{\Functor{F}})_X = \id_{\Functor{F}(X)}$
  for each $X \in \Ob(\Category{C})$.
  Naturality holds since both composites in~\eqref{eq:naturality-square}
  reduce to $\Functor{F}(f)$.
\end{definition}

\begin{definition}[Vertical composition]
  \label{def:vertical-composition}
  \concept{vertical-composition}{Vertical Composition}
  \depends{natural-transformation}
  Given natural transformations
  $\NatTrans{\alpha} \colon \Functor{F} \Rightarrow \Functor{G}$
  and
  $\NatTrans{\beta} \colon \Functor{G} \Rightarrow \Functor{H}$,
  their \newterm{vertical composite}
  $\newmath{\NatTrans{\beta} \circ \NatTrans{\alpha}} \colon \Functor{F} \Rightarrow \Functor{H}$
  has components
  \begin{equation}
    \label{eq:vertical-comp}
    (\NatTrans{\beta} \circ \NatTrans{\alpha})_X = \NatTrans{\beta}_X \circ \NatTrans{\alpha}_X
  \end{equation}
  for each $X \in \Ob(\Category{C})$.
\end{definition}

\begin{proposition}
  \label{prop:vertical-comp-nat}
  The vertical composite $\NatTrans{\beta} \circ \NatTrans{\alpha}$
  as defined in~\eqref{eq:vertical-comp} is a natural transformation
  $\Functor{F} \Rightarrow \Functor{H}$.
\end{proposition}

\begin{proof}
  For any morphism $f \colon X \lto Y$ in $\Category{C}$,
  \[
    \Functor{H}(f) \circ (\NatTrans{\beta} \circ \NatTrans{\alpha})_X
    = \Functor{H}(f) \circ \NatTrans{\beta}_X \circ \NatTrans{\alpha}_X
    = \NatTrans{\beta}_Y \circ \Functor{G}(f) \circ \NatTrans{\alpha}_X
    = \NatTrans{\beta}_Y \circ \NatTrans{\alpha}_Y \circ \Functor{F}(f)
    = (\NatTrans{\beta} \circ \NatTrans{\alpha})_Y \circ \Functor{F}(f),
  \]
  where the second and third equalities use naturality of $\NatTrans{\beta}$
  and $\NatTrans{\alpha}$ respectively.
\end{proof}

\begin{definition}[Horizontal composition / Godement product]
  \label{def:horizontal-composition}
  \concept{horizontal-composition}{Horizontal Composition}
  \depends{natural-transformation}
  \depends{functor}
  Given categories $\Category{C}$, $\Category{D}$, $\Category{E}$,
  functors $\Functor{F}, \Functor{F}' \colon \Category{C} \lto \Category{D}$
  and $\Functor{G}, \Functor{G}' \colon \Category{D} \lto \Category{E}$,
  and natural transformations
  $\NatTrans{\alpha} \colon \Functor{F} \Rightarrow \Functor{F}'$
  and $\NatTrans{\beta} \colon \Functor{G} \Rightarrow \Functor{G}'$,
  the \newterm{horizontal composite} (or \newterm{Godement product})
  $\newmath{\NatTrans{\beta} \star \NatTrans{\alpha}}
    \colon \Functor{G} \circ \Functor{F} \Rightarrow \Functor{G}' \circ \Functor{F}'$
  has components
  \begin{equation}
    \label{eq:horizontal-comp}
    (\NatTrans{\beta} \star \NatTrans{\alpha})_X
    = \NatTrans{\beta}_{\Functor{F}'(X)} \circ \Functor{G}(\NatTrans{\alpha}_X)
    = \Functor{G}'(\NatTrans{\alpha}_X) \circ \NatTrans{\beta}_{\Functor{F}(X)},
  \end{equation}
  for each $X \in \Ob(\Category{C})$.
  The two expressions in~\eqref{eq:horizontal-comp} are equal
  by naturality of $\NatTrans{\beta}$ applied to $\NatTrans{\alpha}_X$.
\end{definition}

\begin{remark}
  Vertical composition is associative and admits $\id_{\Functor{F}}$
  as a two-sided unit.
  Together with the interchange law
  relating vertical and horizontal composition,
  these data make functors and natural transformations
  between two fixed categories into a category
  $\newmath{[\Category{C}, \Category{D}]}$
  (the \newterm{functor category}),
  and all small categories into a 2-category $\Category{Cat}$.
\end{remark}
