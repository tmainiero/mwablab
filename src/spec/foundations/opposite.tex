\section{The Opposite Category}
\label{sec:opposite}

\begin{definition}[Opposite category]
  \label{def:opposite-category}
  Let $\caty{C}$ be a category.
  The \newterm{opposite category} (or \newterm{dual category})
  $\newmath{\caty{C}^{\mathrm{op}}}$ is defined by the following data:
  \begin{enumerate}
    \item $\operatorname{Ob}(\caty{C}^{\mathrm{op}}) = \operatorname{Ob}(\caty{C})$.
    \item For each pair of objects $X, Y$,
      \begin{equation}
        \label{eq:op-hom}
        \operatorname{Hom}_{\caty{C}^{\mathrm{op}}}(X, Y)
        = \operatorname{Hom}_{\caty{C}}(Y, X).
      \end{equation}
    \item For each object $X$,
      the identity morphism is $\newmath{\operatorname{id}^{\mathrm{op}}_X} = \operatorname{id}_X$.
    \item For morphisms
      $f \in \operatorname{Hom}_{\caty{C}^{\mathrm{op}}}(X, Y)$
      and $g \in \operatorname{Hom}_{\caty{C}^{\mathrm{op}}}(Y, Z)$
      (i.e.\ $f \colon Y \to X$ and $g \colon Z \to Y$ in $\caty{C}$),
      the \newterm{opposite composition} is
      \begin{equation}
        \label{eq:op-comp}
        g \circ^{\mathrm{op}} f = f \circ g \in \operatorname{Hom}_{\caty{C}}(Z, X)
        = \operatorname{Hom}_{\caty{C}^{\mathrm{op}}}(X, Z).
      \end{equation}
  \end{enumerate}
\end{definition}

\begin{remark}[Opposite category satisfies the axioms]
  \label{rem:opposite-axioms}
  We verify that $\caty{C}^{\mathrm{op}}$ is indeed a category.
  \begin{itemize}
    \item \textbf{Associativity.}
      For morphisms $f \colon X \to Y$, $g \colon Y \to Z$, $h \colon Z \to W$
      in $\caty{C}^{\mathrm{op}}$
      (i.e.\ $f \colon Y \to X$, $g \colon Z \to Y$, $h \colon W \to Z$ in $\caty{C}$),
      \[
        h \circ^{\mathrm{op}} (g \circ^{\mathrm{op}} f)
        = h \circ^{\mathrm{op}} (f \circ g)
        = (f \circ g) \circ h
        = f \circ (g \circ h)
        = (g \circ h) \circ^{\mathrm{op}} f
        = (h \circ^{\mathrm{op}} g) \circ^{\mathrm{op}} f,
      \]
      using associativity of composition in $\caty{C}$.

    \item \textbf{Identity.}
      For $f \colon X \to Y$ in $\caty{C}^{\mathrm{op}}$
      (i.e.\ $f \colon Y \to X$ in $\caty{C}$),
      \[
        f \circ^{\mathrm{op}} \operatorname{id}^{\mathrm{op}}_X
        = \operatorname{id}_X \circ f = f
        \quad \text{and} \quad
        \operatorname{id}^{\mathrm{op}}_Y \circ^{\mathrm{op}} f
        = f \circ \operatorname{id}_Y = f,
      \]
      using the identity laws in $\caty{C}$.
  \end{itemize}
\end{remark}

\begin{proposition}
  \label{prop:opposite-involution}
  For any category $\caty{C}$, there is an equality
  $(\caty{C}^{\mathrm{op}})^{\mathrm{op}} = \caty{C}$.
\end{proposition}

\begin{proof}
  The objects and hom-sets of $(\caty{C}^{\mathrm{op}})^{\mathrm{op}}$
  are recovered by the double reversal
  \[
    \operatorname{Hom}_{(\caty{C}^{\mathrm{op}})^{\mathrm{op}}}(X, Y)
    = \operatorname{Hom}_{\caty{C}^{\mathrm{op}}}(Y, X)
    = \operatorname{Hom}_{\caty{C}}(X, Y),
  \]
  and composition $f \circ^{\mathrm{op\,op}} g = g \circ^{\mathrm{op}} f = f \circ g$
  agrees with composition in $\caty{C}$.
\end{proof}

\begin{definition}[Contravariant functor]
  \label{def:contravariant-functor}
  A \newterm{contravariant functor} from $\caty{C}$ to $\caty{D}$
  is a \newterm{(covariant) functor}
  $\fun{F} \colon \caty{C}^{\mathrm{op}} \to \caty{D}$.

  Explicitly, $\fun{F}$ assigns to each object $X \in \operatorname{Ob}(\caty{C})$
  an object $\fun{F}(X) \in \operatorname{Ob}(\caty{D})$,
  and to each morphism $f \colon X \to Y$ in $\caty{C}$
  a morphism
  \begin{equation}
    \label{eq:contravariant-hom}
    \fun{F}(f) \colon \fun{F}(Y) \to \fun{F}(X)
  \end{equation}
  in $\caty{D}$,
  subject to $\fun{F}(\operatorname{id}_X) = \operatorname{id}_{\fun{F}(X)}$
  and $\fun{F}(g \circ f) = \fun{F}(f) \circ \fun{F}(g)$.
\end{definition}

\begin{remark}
  The representable presheaf functor
  $\operatorname{Hom}_{\caty{C}}(-, Y) \colon \caty{C}^{\mathrm{op}} \to \Set$
  is the canonical example of a contravariant functor.
  The collection of all such functors organises into the presheaf category
  $\newmath{[\caty{C}^{\mathrm{op}}, \Set]}$,
  into which $\caty{C}$ embeds fully faithfully
  via the Yoneda embedding.
\end{remark}
