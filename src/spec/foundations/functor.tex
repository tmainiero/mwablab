\section{Functors}
\label{sec:functor}

% Typing macros
\providecommand{\caty}[1]{\mathsf{#1}}

\providecommand{\nat}[1]{\mathsf{#1}}
\providecommand{\newmath}[1]{%
  {\setlength{\fboxsep}{0pt}%
   \mathchoice
     {\colorbox{defblue}{$\displaystyle #1$}}

Let $\caty{C}$ and $\caty{D}$ be categories.
A \newterm{functor} $\newmath{\fun{F} \colon \caty{C} \to \caty{D}}$ consists of the following data:
\begin{enumerate}
  \item A map
    $\fun{F} \colon \operatorname{Ob}(\caty{C}) \to \operatorname{Ob}(\caty{D})$
    on objects.
  \item For each pair of objects $X, Y \in \operatorname{Ob}(\caty{C})$,
    a map on hom-sets
    \begin{equation}
      \label{eq:functor-hom}
      \fun{F}_{X,Y} \colon
        \operatorname{Hom}_{\caty{C}}(X, Y)
        \to
        \operatorname{Hom}_{\caty{D}}(\fun{F}(X), \fun{F}(Y)).
    \end{equation}
\end{enumerate}

subject to the following axioms:
\begin{itemize}
  \item \textbf{Preservation of identities.}
    For each object $X \in \operatorname{Ob}(\caty{C})$,
    \begin{equation}
      \label{eq:functor-id}
      \fun{F}(\operatorname{id}_X) = \operatorname{id}_{\fun{F}(X)}.
    \end{equation}

  \item \textbf{Preservation of composition.}
    For all morphisms $f \colon X \to Y$ and $g \colon Y \to Z$ in $\caty{C}$,
    \begin{equation}
      \label{eq:functor-comp}
      \fun{F}(g \circ f) = \fun{F}(g) \circ \fun{F}(f).
    \end{equation}
\end{itemize}

\begin{definition}[Identity functor]
  \label{def:identity-functor}
  The \newterm{identity functor} on $\caty{C}$ is
  $\newmath{\operatorname{Id}_{\caty{C}}} \colon \caty{C} \to \caty{C}$,
  defined by $\operatorname{Id}_{\caty{C}}(X) = X$ on objects
  and $\operatorname{Id}_{\caty{C}}(f) = f$ on morphisms.
  It satisfies both functor axioms trivially.
\end{definition}

\begin{definition}[Composition of functors]
  \label{def:functor-composition}
  Given functors $\fun{F} \colon \caty{C} \to \caty{D}$
  and $\fun{G} \colon \caty{D} \to \caty{E}$,
  their \newterm{composite}
  $\newmath{\fun{G} \circ \fun{F}} \colon \caty{C} \to \caty{E}$
  is defined by
  \begin{align}
    \label{eq:functor-comp-ob}
    (\fun{G} \circ \fun{F})(X) &= \fun{G}(\fun{F}(X))
      \quad \text{on objects,} \\
    \label{eq:functor-comp-mor}
    (\fun{G} \circ \fun{F})(f) &= \fun{G}(\fun{F}(f))
      \quad \text{on morphisms.}
  \end{align}
\end{definition}

\begin{proposition}
  \label{prop:functor-comp-is-functor}
  The composite $\fun{G} \circ \fun{F}$ as defined above is a functor.
\end{proposition}

\begin{proof}
  Preservation of identities:
  \[
    (\fun{G} \circ \fun{F})(\operatorname{id}_X)
    = \fun{G}(\fun{F}(\operatorname{id}_X))
    = \fun{G}(\operatorname{id}_{\fun{F}(X)})
    = \operatorname{id}_{\fun{G}(\fun{F}(X))}
    = \operatorname{id}_{(\fun{G} \circ \fun{F})(X)}.
  \]
  Preservation of composition: for $f \colon X \to Y$ and $g \colon Y \to Z$,
  \[
    (\fun{G} \circ \fun{F})(g \circ f)
    = \fun{G}(\fun{F}(g \circ f))
    = \fun{G}(\fun{F}(g) \circ \fun{F}(f))
    = \fun{G}(\fun{F}(g)) \circ \fun{G}(\fun{F}(f))
    = (\fun{G} \circ \fun{F})(g) \circ (\fun{G} \circ \fun{F})(f). \qedhere
  \]
\end{proof}

\begin{remark}
  Functor composition is associative,
  and $\operatorname{Id}_{\caty{C}}$ is a two-sided unit.
  Consequently, small categories and functors form a category $\newmath{\caty{Cat}}$.
\end{remark}
