% --- Typing macros -----------------------------------------------------------
% CamelCase: the macro name IS the type. \Category{C} means "C is a category."

\newcommand{\Functor}[1]{\mathsf{#1}}        % functors
\newcommand{\NatTrans}[1]{\mathsf{#1}}       % natural transformations
\newcommand{\Monoidal}[1]{\mathsf{#1}}       % monoidal categories
\newcommand{\Enrichment}[1]{\mathsf{#1}}     % enrichment bases

% Common instances (derived from type macros)
\newcommand{\Set}{\Category{Set}}

% --- Arrows (auto long/short via mathchoice) ---------------------------------
% \lto    : ->  in text, --> in display
% \lTo    : =>  in text, ==> in display
% \lmapsto: |-> in text, |--> in display
\newcommand{\lto}{\mathchoice%
  {\longrightarrow}%   displaystyle
  {\rightarrow}%       textstyle
  {\rightarrow}%       scriptstyle
  {\rightarrow}%       scriptscriptstyle
}
\newcommand{\lTo}{\mathchoice%
  {\Longrightarrow}%
  {\Rightarrow}%
  {\Rightarrow}%
  {\Rightarrow}%
}
\newcommand{\lmapsto}{\mathchoice%
  {\longmapsto}%
  {\mapsto}%
  {\mapsto}%
  {\mapsto}%
}

% --- New symbol/term highlighting --------------------------------------------
\newcommand*{\newmath}[1]{%
  {\setlength{\fboxsep}{0pt}%
   \mathchoice
     {\colorbox{defblue}{$\displaystyle #1$}}
     {\colorbox{defblue}{$\textstyle #1$}}
     {\colorbox{defblue}{$\scriptstyle #1$}}
     {\colorbox{defblue}{$\scriptscriptstyle #1$}}}}
\newcommand{\newterm}[1]{\textcolor{blue}{\emph{#1}}}

% --- Semantic preprocessor macros (semtex) -----------------------------------
% These are transparent to pdflatex. The semtex preprocessor extracts them.
% \concept{id}{Display Name} — declares a concept defined in this section.
\newcommand{\concept}[2]{\label{concept:#1}}
% \depends{concept-id} — this definition depends on concept-id.
\newcommand{\depends}[1]{}
% \implements{lang}{module-or-path} — cross-language implementation link.
\newcommand{\implements}[2]{}
% \axiom{concept-id}{axiom-name} — names a law/axiom within a concept.
\newcommand{\axiom}[2]{\label{axiom:#1:#2}}
% \uses{concept-id} — weak reference (mentions, but does not depend on).
\newcommand{\uses}[1]{}
% \stacksref{tag} — Stacks Project reference with link.
\newcommand{\stacksref}[1]{%
  \textsc{Stacks Project} Tag
  \href{https://stacks.math.columbia.edu/tag/#1}{\texttt{#1}}%
}
% \nlabref{page} — nLab reference with link.
\newcommand{\nlabref}[1]{%
  \href{https://ncatlab.org/nlab/show/#1}{nLab: \textsf{#1}}%
}
