% --- Typing macros -----------------------------------------------------------
% These add a semantic layer: \caty{C} means "C is a category", etc.
% Even when formatting is identity, the semantic tag helps readability and
% makes global style changes trivial.

\newcommand{\fun}[1]{\mathsf{#1}}        % functors
\newcommand{\nat}[1]{\mathsf{#1}}        % natural transformations
\newcommand{\mon}[1]{\mathsf{#1}}        % monoidal categories
\newcommand{\venr}[1]{\mathsf{#1}}       % enrichment bases
